44 inline float noise(float x, float y, unsigned int _seed)
45 {
46 std::vector<int> p;
47
48 p.resize(256);
49 std::iota(p.begin(), p.end(), 0);
50
51 std::default_random_engine engine(_seed);
52 std::shuffle(p.begin(), p.end(), engine);
53
54 // Duplicate to avoid wrapping
55 p.insert(p.end(), p.begin(), p.end());
56
57 // Find unit grid cell containing point
58 int X = (int)floor(x) & 255;
59 int Y = (int)floor(y) & 255;
60
61 // Relative coords inside cell
62 x -= floor(x);
63 y -= floor(y);
64
65 // Fade curves
66 float u = fade(x);
67 float v = fade(y);
68
69 // Hash coords of the 4 corners
70 int aa = p[p[X ] + Y ];
71 int ab = p[p[X ] + Y + 1];
72 int ba = p[p[X + 1] + Y ];
73 int bb = p[p[X + 1] + Y + 1];
74
75 // Blend results
76 float res = lerp(v,
77 lerp(u, grad(aa, x, y), grad(ba, x - 1, y)),
78 lerp(u, grad(ab, x, y - 1), grad(bb, x - 1, y - 1))
79 );
80 return (res + 1.0f) * 0.5f; // normalize to [0,1]
81 }
